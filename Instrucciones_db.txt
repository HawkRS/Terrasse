---------------------------------------------------------------------------------------------
crearemos las tablas correspondientes
mysql> create table registro (nombre varchar(50) not null, username varchar(20) not null primary key, telefono varchar(15) not null, cel varchar(15), domicilio varchar(30) not null, colonia varchar(20) not null, cp int(6) not null, email varchar(30) not null, pass varchar(20) not null);

mysql> create table pedido (id_pedido int not null primary key auto_increment, username varchar(20) not null, fecha date not null, id_articulo int not null, cantidad int not null, total double not null);

mysql> create table producto (id_producto int not null primary key auto_increment, foto varchar(20) not null, descripcion text not null, precio double not null);
